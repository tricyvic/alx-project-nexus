Database
1. pip install psycopg2-binary



2. Update settings.py
DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.postgresql',
        'NAME': 'nexus_db',
        'USER': 'nexus_user',
        'PASSWORD': 'nexus_pass',
        'HOST': 'localhost',  # or use '127.0.0.1'
        'PORT': '5432',
    }
}

# Access PostgreSQL shell
sudo -u postgres psql

# Create the database
CREATE DATABASE nexus_db;

# Create the user
CREATE USER nexus_user WITH PASSWORD 'nexus_pass';

# Grant privileges
GRANT ALL PRIVILEGES ON DATABASE nexus_db TO nexus_user;

# Exit psql
\q


sudo -u postgres psql -d nexus_db

GRANT USAGE ON SCHEMA public TO nexus_user;
GRANT CREATE ON SCHEMA public TO nexus_user;
GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA public TO nexus_user;
GRANT ALL PRIVILEGES ON ALL SEQUENCES IN SCHEMA public TO nexus_user;
